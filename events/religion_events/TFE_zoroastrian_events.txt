namespace = TFE_zoroastrian


TFE_zoroastrian.0001 = {
	hidden = yes
	
	immediate = {
		create_character = {
			location = 4053
			#General
			name = "Yazd"
			gender = male
			age = 32
			faith = faith:mazdakism
			culture = culture:persian
			dynasty = generate
			#Education
			trait = education_learning_4
			trait = compassionate
			trait = honest
			trait = just
			trait = intellect_good_1
			trait = theologian
			stewardship = 5
			learning = 10
			martial = 3
			diplomacy = 10
			intrigue = 10
			save_scope_as = mazdak
		} 
		title:d_mazdakism = {
			create_title_and_vassal_change = {
				type = created
				save_scope_as = change
				add_claim_on_loss = no
			}
			change_title_holder = {
				holder = scope:mazdak
				change = scope:change
			}
			resolve_title_and_vassal_change = scope:change
		}
		faith:mazdakism = {
			change_fervor = {
				value = 5
			}
		}
		hidden_effect = {
			faith:mazdakism = {
				remove_doctrine = unavailable_doctrine
			}
		}
		every_ruler = {
			limit = {
				AND = {
					religion = religion:zoroastrianism_religion
					top_liege = {
						has_title = title:e_persia
					}
				}
			}
			trigger_event = {
				id = TFE_zoroastrian.0002
				days = {30 50}
			}
		}
	}
}

TFE_zoroastrian.0002 = {
	type = character_event
	title = TFE_zoroastrian.0001.t
	desc = TFE_zoroastrian.0001.desc
	theme = faith

	right_portrait = {
		character = scope:mazdak
		animation = personality_zealous
	}

	# Embrace Mazdakism
	option = {
		name = TFE_zoroastrian.0001.a
		faith:mazdakism = {
			change_fervor = {
				value = 5
			}
		}
		set_character_faith_with_conversion = faith:mazdakism
		
	}
	#Embrace Mazdakism and Invite him to Court
	option = {
		name = TFE_zoroastrian.0001.b
		add_courtier = scope:mazdak
		trigger = {
			has_title = title:c_baghdad
		}
		faith:mazdakism = {
			change_fervor = {
				value = 5
			}
		}
		set_character_faith_with_conversion = faith:mazdakism
		
	}
	# What Mockery of our faith is this?
	option = {
		name = TFE_zoroastrian.0001.c
	}
}

TFE_zoroastrian.0003 = {
	hidden = yes
	
	immediate = {
		faith:khurramism = {
			change_fervor = {
				value = 5
			}
		}
		set_global_variable = {
			name = khurramism_spawned
			value = yes
		}
		hidden_effect = {
			faith:khurramism = {
				remove_doctrine = unavailable_doctrine
			}
		}
		every_county_in_region = {
			region = ghw_region_persia
			random = {
				chance = 5
				set_county_faith = faith:khurramism
				if = {
					limit = {
						holder = {
							NOT = {
								religion = religion:zoroastrianism_religion
							}
						}

					}
					add_county_modifier = {
						modifier = rebellious_people
						years = 10
					}
				}
			}
		}
		every_ruler = {
			limit = {
				AND = {
					capital_province = {
						geographical_region = ghw_region_persia
					}
				}
			}
			trigger_event = {
				id = TFE_zoroastrian.0004
				days = {30 50}
			}
		}
	}
}

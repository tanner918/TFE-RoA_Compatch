province_events.0014 = {
	title = province_events.0014.t
	desc = province_events.0014.desc
	scope = character
	theme = martial
	override_background = {
		reference = throne_room 
	}
	right_portrait = {
		character = title:c_viterbo.holder
		animation = personality_honorable
	}

	immediate = {
		play_music_cue = "TFE_General_Cue"
	}

	option = {
		name = province_events.0014.option.a
		add_prestige = -200
		save_temporary_scope_as = tmp_usurp
		top_liege = {
			save_temporary_scope_as = old_liege
			every_vassal = {
				add_to_global_variable_list = {
					name = vassal_usurp
					target = THIS
				}
				create_title_and_vassal_change = {
					type = usurped
					save_scope_as = vassal_change
					add_claim_on_loss = yes
				}
				change_liege = {
					liege = scope:tmp_usurp
					change = scope:vassal_change
				}
				
				resolve_title_and_vassal_change = scope:vassal_change
				
			}
			every_held_title = {
				limit = {
					tier = tier_empire
				}
				create_title_and_vassal_change = {
					type = usurped
					save_scope_as = title_change
					add_claim_on_loss = yes
				}
				THIS = {
					save_scope_as = title_liege_change
				}
				scope:title_liege_change = {
					change_title_holder = {
						holder = scope:tmp_usurp
						change = scope:title_change
					}
				}
				
				resolve_title_and_vassal_change = scope:title_change
			}
		}
		create_title_and_vassal_change = {
			type = usurped
			save_scope_as = vassal_change
			add_claim_on_loss = yes
		}
		scope:old_liege = {
			change_liege = {
				liege = scope:tmp_usurp
				change = scope:vassal_change
			}
		}
		
		resolve_title_and_vassal_change = scope:vassal_change
		every_in_global_list = {
			variable = vassal_usurp
			create_title_and_vassal_change = {
				type = usurped
				save_scope_as = vassal_change
				add_claim_on_loss = yes
			}
			change_liege = {
				liege = scope:tmp_usurp
				change = scope:vassal_change
			}
			
			resolve_title_and_vassal_change = scope:vassal_change
		}
		if = {
					limit = {
						has_dlc_feature = roads_to_power
					}
						change_government = administrative_government
					}
				else = {
						change_government = imperial_government
					}
    }
	option = {
		name = province_events.0014.option.b
		custom_tooltip = province_events.0014.option.b.tooltip
		top_liege = {
			every_vassal = {
				limit = {
					NOT = {
						THIS = PREV
						THIS = title:c_viterbo.holder
					}
					highest_held_title_tier >= tier_county
				}
				trigger_event = {
					id = province_events.0015
					days = 1
				}
			}
			trigger_event = {
				id = province_events.0016
				days = 10
			}
		}
    }
}

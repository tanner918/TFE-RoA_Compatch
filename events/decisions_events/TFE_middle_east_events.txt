	type = character_event
	title = TFE_middle_east.0001.t
	theme = war
	desc = {

			triggered_desc = {

					var:TFE_regional_units_eran = flag:daylami

	}
	
	right_portrait = {
		character = root
		animation = personality_rational
	}
	
	# Pay basic amount

			#OR = {
			#	is_ai = no
			#	gold >= 500
			#}
		#}
		remove_short_term_gold = 500
		if = {
			limit = { var:TFE_regional_units_eran = flag:war_elephants } 
			spawn_army = {
				name = event_troop_default_name
				men_at_arms = {
					type = war_elephant
					stacks = 2
				}
				location = capital_province
				origin = capital_province
				uses_supply = yes
				inheritable = yes
			}
		}
		else = {
			spawn_army = {
				name = event_troop_default_name
				levies = 250
				men_at_arms = {
					type = daylami
					stacks = 5
				}
				location = capital_province
				origin = capital_province
				uses_supply = yes
				inheritable = yes
			}
		}
	}
	# Pay medium amount
	option = {
		name = TFE_middle_east.0001.b
		trigger = {
			OR = {
				is_ai = no
				gold >= 1000
			}
		}
		remove_short_term_gold = 1000
		if = {
			limit = { var:TFE_regional_units_eran = flag:war_elephants }
			spawn_army = {
				name = event_troop_default_name
				men_at_arms = {
					type = war_elephant
					stacks = 4
				}
				location = capital_province
				origin = capital_province
				uses_supply = yes
				inheritable = yes
			}
		}
		else = {
			spawn_army = {
				name = event_troop_default_name
				levies = 500
				men_at_arms = {
					type = daylami
					stacks = 10
				}
				location = capital_province
				origin = capital_province
				uses_supply = yes
				inheritable = yes
			}
		}
	}
	# Pay large amount
	option = {
		name = TFE_middle_east.0001.c
		trigger = {
			OR = {
				is_ai = no
				gold >= 2000
			}
		}
		remove_short_term_gold = 2000
		if = {
			limit = { var:TFE_regional_units_eran = flag:war_elephants }
			spawn_army = {
				name = event_troop_default_name
				men_at_arms = {
					type = war_elephant
					stacks = 8
				}
				location = capital_province
				origin = capital_province
				uses_supply = yes
				inheritable = yes
			}
		}
		else = {
			spawn_army = {
				name = event_troop_default_name
				levies = 1000
				men_at_arms = {
					type = daylami
					stacks = 20
				}
				location = capital_province
				origin = capital_province
				uses_supply = yes
				inheritable = yes
			}
		}
	}

	after = {
		if = {
			limit = { has_variable = TFE_regional_units_eran }
			remove_variable = TFE_regional_units_eran
		}
	}

}

language_anglic = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_anglic
		}
	}
	parameters = {
		language_group_germanic = yes
		language_family_indo_european = yes 		
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_anglic }
			multiply = 10
		}
		else_if = {
			limit = { has_cultural_parameter = language_group_germanic }
			multiply = 5
		}
		else_if = {
			limit = { has_cultural_parameter = language_family_indo_european }
			multiply = 2.5
		}
	}
	
	color = english
}

language_chinese = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_chinese
		}
	}
	parameters = {
		language_group_sinitic = yes
		language_family_sino_tibetan = yes 		
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_chinese }
			multiply = 10
		}
		else_if = {
			limit = { has_cultural_parameter = language_group_sinitic }
			multiply = 5
		}
		else_if = {
			limit = { has_cultural_parameter = language_family_sino_tibetan }
			multiply = 2.5
		}
		if = {
			limit = {
				OR = {
					culture = culture:xiongnu
					culture = culture:xianbei
					culture = culture:khitan
					culture = {
						any_parent_culture_or_above = {
							OR = {
								this = culture:xiongnu
								this = culture:xianbei
								this = culture:khitan
							}
						}
					}
				}				
			}
			multiply = 2.5
		}
		if = {
			limit = {
				scope:character = {
					religion = religion:taoism_religion
				}			
			}
			multiply = 2.5
		}
		if = {
			limit = {
				scope:character = {
					capital_province = {
						geographical_region = custom_liang
					}
				}			
			}
			multiply = 5
		}
	}
	
	color = han
}

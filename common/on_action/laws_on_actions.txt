on_game_start = {
	on_actions = {
		on_laws_start
	}
}

on_laws_start = {
	effect = {
		every_ruler = {
			if = {
				limit = {
					has_government = administrative_government
					primary_title = {
						has_title_law = imperial_elective_succession_law
					}
				}
				primary_title = {
					clear_title_laws = yes
				}
			}
			limit = {
				has_imperial_laws = yes
				OR = {
					has_game_rule = imperial_competence_enabled
					has_game_rule = imperial_competence_enabled_in_430
				}
			}
			trigger_event = {
				id = collapse.0005
				days = 1
			}
			set_variable = {
				name = mil_comp
				value = 1500
			}
			set_variable = {
				name = civic_comp
				value = 1500
			}
			set_variable = {
				name = diplo_comp
				value = 1500
			}
			
		}
	}
	on_actions = {
		delay = { days = { 365 420 } }
		on_yearly_mil_laws
		on_yearly_civic_laws
		on_yearly_diplo_laws
		on_yearly_partial_collapse_check
	}
}

on_yearly_mil_laws = {
	effect = {
		every_ruler = {
			if = {
				limit = {
					AND = {
						has_imperial_laws = yes
						NOT = {exists = var:mil_comp}
						OR = {
							has_game_rule = imperial_competence_enabled
							has_game_rule = imperial_competence_enabled_in_430
						}
					}
				}
				set_variable = {
					name = mil_comp
					value = 1500
				}
			}
			if = {
				limit = {
					AND = {
						has_imperial_laws = yes
						var:mil_comp = 0
						var:civic_comp = 0
						var:diplo_comp = 0
						OR = {
							has_title = title:e_byzantium
							has_title = title:e_western_roman_empire
							has_title = title:e_roman_empire
						}
						OR = {
							has_game_rule = imperial_competence_enabled
							AND = {
								has_game_rule = imperial_competence_enabled_in_430
								current_date >= 430.1.1
							}
						}
					}
				}
				trigger_event = collapse.0001
				
			}

			if = {
				limit = {
					AND = {
						exists = var:mil_comp
						NOT = {
							var:mil_comp = 0
						}
						has_imperial_laws = yes
						OR = {
							has_game_rule = imperial_competence_enabled
							AND = {
								has_game_rule = imperial_competence_enabled_in_430
								current_date >= 430.1.1
							}
						}
					}
				}
				change_variable = {
					name = mil_comp
					add = mil_competence_net
				}
				clamp_variable = {
					name = mil_comp
					max = 6000
					min = 0
				}
				if = {
					limit = {
						AND = {
							NOT = {
								var:mil_comp = 0
							}
							has_title = title:e_byzantium
							any_character_struggle = {
								involvement = involved
								is_struggle_type = roman_persian_struggle	
							}
							var:mil_comp >= 1500
						}
					}
					struggle:roman_persian_struggle = {
						activate_struggle_catalyst = { catalyst = catalyst_high_mil_competence character = prev }
					}
				}
				else_if = {
					limit = {
						AND = {
							NOT = {
								var:mil_comp = 0
							}
							has_title = title:e_byzantium
							any_character_struggle = {
								involvement = involved
								is_struggle_type = roman_persian_struggle	
							}
							var:mil_comp <= 1000
						}
					}
					struggle:roman_persian_struggle = {
						activate_struggle_catalyst = { catalyst = catalyst_low_mil_competence character = prev }
					}
				}
				if = {
					limit = {
						AND = {
							var:mil_comp = 0
						}
					}
					trigger_event = collapse.0002
				}
				if = {
					limit = {
						AND = {
							has_game_rule = imperial_competence_collapse_enabled
							var:civic_comp = 0
							var:mil_comp = 0
							var:diplo_comp = 0
						}
					}
					trigger_event = collapse.0001
				}
			}

			

			
		}
	}
	on_actions = {
		delay = { days = { 365 420 } }
		on_yearly_mil_laws
	}
}

on_yearly_partial_collapse_check = {
	effect = {
		if = {
			limit = {
				AND = {
					title:e_western_roman_empire = {
						is_title_created = yes
					}
					title:e_western_roman_empire.holder ?= {
						NOT = {
							exists = global_var:wre_partial_collapsed
						}
						AND = {
							exists = var:mil_comp
							exists = var:civic_comp
							exists = var:diplo_comp
						}
						has_imperial_laws = yes
						OR = {
							has_game_rule = imperial_competence_enabled
							AND = {
								has_game_rule = imperial_competence_enabled_in_430
								current_date >= 430.1.1
							}
						}
						has_game_rule = imperial_competence_partial_collapse_enabled
					}
				}
			}
			if = {
				limit = {
					title:e_western_roman_empire.holder ?= {
						OR = {
							AND = {
								var:diplo_comp = 0
								var:civic_comp = 0
							}
							AND = {
								var:diplo_comp = 0
								var:mil_comp = 0
							}
							AND = {
								var:civic_comp = 0
								var:mil_comp = 0
							}
						}
					}
				}
				title:e_western_roman_empire.holder ?= {
					trigger_event = collapse.0010
				}
			}
		}
		if = {
			limit = {
				AND = {
					title:e_byzantium = {
						is_title_created = yes
					}
					title:e_byzantium.holder ?= {
						NOT = {
							exists = global_var:ere_partial_collapsed
						}
						AND = {
							exists = var:mil_comp
							exists = var:civic_comp
							exists = var:diplo_comp
						}
						has_imperial_laws = yes
						OR = {
							has_game_rule = imperial_competence_enabled
							AND = {
								has_game_rule = imperial_competence_enabled_in_430
								current_date >= 430.1.1
							}
						}
						has_game_rule = imperial_competence_partial_collapse_enabled
					}
				}
			}
			if = {
				limit = {
					title:e_byzantium.holder ?= {
						OR = {
							AND = {
								var:diplo_comp = 0
								var:civic_comp = 0
							}
							AND = {
								var:diplo_comp = 0
								var:mil_comp = 0
							}
							AND = {
								var:civic_comp = 0
								var:mil_comp = 0
							}
						}
					}
				}
				title:e_byzantium.holder ?= {
					trigger_event = collapse.0010
				}
			}
		}
	}
	on_actions = {
		delay = { days = { 365 420 } }
		on_yearly_partial_collapse_check
	}
}

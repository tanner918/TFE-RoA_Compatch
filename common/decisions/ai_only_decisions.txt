support_the_blues = {
	ai_check_interval = 120
	ai_will_do = 70

	is_shown = {
		is_ruler = yes
		is_landed_or_landless_administrative = yes
		is_ai = yes
		AND = {
			NOT = {has_trait = blue_support}
			NOT = {has_trait = green_support}
		}
	OR = {
		government_has_flag = government_is_imperial
		government_has_flag = government_is_administrative
		}
	}

	effect = {
		add_trait = blue_support
	}
}

zakah_dev_realm = {
	ai_check_interval = 120
	ai_will_do = 70

	is_shown = {
		is_ruler = yes
		is_landed_or_landless_administrative = yes
		is_ai = yes
		exists = var:zakah_balance
		var:zakah_balance >= 500
	}

	effect = {
		change_variable = {
			name = zakah_balance
			subtract = 500
		}

		set_variable = {
			name = dev_zakah_tmp_count
			value = 50
		}
		change_variable = {
			name = dev_zakah_tmp_count
			divide = realm_size
		}
		every_realm_county = {
			change_development_level = root.var:dev_zakah_tmp_count
		}
	}
}

zakah_dev_all = {
	ai_check_interval = 120
	ai_will_do = 70

	is_shown = {
		is_ruler = yes
		is_landed_or_landless_administrative = yes
		is_ai = yes
		exists = var:zakah_balance
		var:zakah_balance >= 1500
	}

	effect = {
		change_variable = {
			name = zakah_balance
			subtract = 1500
		}

		set_variable = {
			name = dev_zakah_tmp_count
			value = 75
		}
		set_variable = {
			name = faith_counties
			value = 0
		}
		hidden_effect = {
			every_county = {
				limit = {
					faith = root.faith
				}
				root = {
					change_variable = {
						name = faith_counties
						add = 1
					}
				}
			}
			change_variable = {
				name = dev_zakah_tmp_count
				divide = var:faith_counties
			}
			every_county = {
				limit = {
					faith = root.faith
				}
				change_development_level = root.var:dev_zakah_tmp_count
			}
		}
		
	}
}

increase_node_cap = {
	ai_check_interval = 120
	ai_will_do = 70

	is_shown = {
		is_ruler = yes
		is_landed_or_landless_administrative = yes
		is_ai = yes
		any_realm_province = {
			exists = var:node
		}
		gold > 500
	}

	effect = {
		remove_short_term_gold = 500
		random_realm_province = {
			limit = {
				exists = var:node
			}
			change_variable = {
				name = trade_node_cap
				add = 5
			}
		}
		
	}
}

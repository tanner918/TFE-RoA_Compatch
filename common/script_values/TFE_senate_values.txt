	value = 0
	title:c_roma.holder.top_liege = {
		save_temporary_scope_as = emperor

		limit = {
			opinion = {
				target = scope:emperor
				value >= 90
			}
		}
		add = 5
	}
	else_if = {
		limit = {
			opinion = {
				target = scope:emperor
				value >= 70
			}
		}
		add = 4
	}
	else_if = {
		limit = {
			opinion = {
				target = scope:emperor
				value >= 50
			}
		}
		add = 3
	}
	else_if = {
		limit = {
			opinion = {
				target = scope:emperor
				value >= 30
			}
		}
		add = 2
	}
	else_if = {
		limit = {
			opinion = {
				target = scope:emperor
				value >= 10
			}
		}
		add = 1
	}

	if = {
		limit = {
			opinion = {
				target = scope:emperor
				value <= -90
			}
		}
		add = -5
	}
	else_if = {
		limit = {
			opinion = {
				target = scope:emperor
				value <= -70
			}
		}
		add = -4
	}
	else_if = {
		limit = {
			opinion = {
				target = scope:emperor
				value <= -50
			}
		}
		add = -3
	}
	else_if = {
		limit = {
			opinion = {
				target = scope:emperor
				value <= -30
			}
		}
		add = -2
	}
	else_if = {
		limit = {
			opinion = {
				target = scope:emperor
				value <= -10
			}
		}
		add = -1
	}

	if = {
		limit = {
			has_trait =high_senator  
		}
		multiply = 3
	}
	else_if = {
		limit = {
			has_trait = senator  
		}
		multiply = 2
	}
}
